12 /**
13 * Threadsafe queue.
14 *
15 * Invariants:
16 * - tail_->value_ == nullptr
17 * - tail->next_ == nullptr
18 * - head_ == tail_ implies an empty list.
19 * - A single element list has head_->next_ == tail_.
20 * - For each node x in the queue, where x!=tail_, x->value_ points to an instance of T and x->next_ points to the next node in the queue.
21 * - x->next_==tail implies x is the last node in the queue.
22 * - Traversing head_->next_ will eventually lead to tail_.
23 *
24 * Addtional Requirements:
25 * - threadsafe_queue must be able to support type T where T is a non-copyable or non-movable type.
26 * - threadsafe_queue does not have to support a non-copyable AND non-movable type.
27 *
28 * Additional Notes:
29 * - threadsafe_queue is non-copyable AND non-movable. (As it has non-copyable and non-movable members.)
30 *
31 * @tparam T The typename of the contained values.
32 */
33 template<typename T>

43 * 2. T has direct or virtual base class that cannot be moved (has deleted, inaccessible, or ambiguous move constructors);
44 * 3. T has direct or virtual base class with a deleted or inaccessible destructor;
45 * 4. T is a union-like class and has a variant member with non-trivial move constructor.

92 {
93 // Construct a dummy node.
94 std::unique_ptr<node> dummy_node = std::make_unique<node>();
95 // Lock the tail mutex.
96 std::lock_guard<mutex_type> lock(tail_mutex_);
97 // Set the tail's value to the new value.
98 tail_->value_ = _value;
99 // Set the tail's next node to the dummy node.
100 tail_->next_ = std::move(dummy_node);
101 // Set the dummy node as the tail node.
102 tail_ = tail_->next_.get();
103 // Increase the element counter.
104 num_elements_++;
105 }

107 /**
108 * Internal function to pop a value from the head of the queue.
109 * @return Pops and return the value from the head of the queue.
110 */
111 std::shared_ptr<T> do_pop()
112 {
113 // Extract the value of the head node.
114 std::unique_ptr<node> old_head = std::move(head_);
115 // Point the head node to it's next node.
116 head_ = std::move(old_head->next_);
117 // Decrease the element counter.
118 num_elements_--;
119 // Return the extracted value.
120 return old_head->value_;
121 }

129 // Lock the mutexes to prevent anyone from pushing.

131 std::lock_guard<mutex_type> tail_lock(_threadsafe_queue->tail_mutex_);
132 // Set the element counter.
133 num_elements_ = _threadsafe_queue->num_elements_.load();
134 // Copy Nodes
135 head_ = _threadsafe_queue->head_ ? std::move(_threadsafe_queue->head_->copy()) : nullptr;
136 tail_ = head_.get();
137 while (tail_ && tail_->next_) { tail_ = tail_->next_.get(); }
138 }
