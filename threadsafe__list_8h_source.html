14 /**
15 * Threadsafe list.
16 *
17 * Invariants:
18 * - Traversing head_->next_ will eventually lead to the bottom node.
19 * - For each node x in the queue, where x!=head_, x->value_ points to an instance of T.
20 * - For each node x in the queue, where x!=tail_, x->next_ points to the next node in the list.
21 * - For a node x, x->next_ == null means that it is the last node.
22 * - head_->next_ == null means the list is empty.
23 *
24 * Addtional Requirements:
25 * - threadsafe_list must be able to support type T where T is a non-copyable or non-movable type.
26 * - threadsafe_list does not have to support a non-copyable AND non-movable type.
27 *
28 * @tparam T The typename of the contained values.
29 */
30 template<typename T>
31 class threadsafe_list : public container {
32 private:

34 typedef std::unique_lock<mutex_type> writer_lock;
35 typedef std::shared_lock<mutex_type> reader_lock;

37 /**
38 * A node containing a value, and the next node in the stack.
39 *
40 * @warning The implicitly-declared or defaulted move constructor for class T is defined as deleted if any of the following is true:
41 * 1. T has non-static data members that cannot be moved (have deleted, inaccessible, or ambiguous move constructors);
42 * 2. T has direct or virtual base class that cannot be moved (has deleted, inaccessible, or ambiguous move constructors);
43 * 3. T has direct or virtual base class with a deleted or inaccessible destructor;
44 * 4. T is a union-like class and has a variant member with non-trivial move constructor.
45 */
46 struct node {
47 // Mutex.

49 // Value.
50 std::shared_ptr<T> value_;

53 };

55 /**
56 * Internal function to add a new value to the front of the list.
57 * @param _value The value to add to the front of the list.
58 * @warning If multiple threads call push_front at the same time, the value will be inserted but may no longer be at the front of the list.

61 {
62 // Construct a new node.
63 std::unique_ptr<node> new_node = std::make_unique<node>();
64 // Set the new node's value to the new value.
65 new_node->value_ = _value;
66 // Lock the head mutex.
67 writer_lock lock(head_.mutex_);
68 // Set new node's next node to the head's next node.
69 new_node->next_ = std::move(head_.next_);
70 // Set the head's next node to the new node.
71 head_.next_ = std::move(new_node);
72 // Increase element count.
73 ++num_elements_;
74 }

127 /**
128 * Checks if any of the values in the list passes the predicate.
129 * @tparam Predicate The typename of the predicate.
130 * @param _predicate Predicate to test the values.
131 * @return Returns true if any of the values in the list passes the predicate. Else, returns false.
132 */
133 template<class Predicate>
134 bool match_any(Predicate&& _predicate) const
135 requires mkr::is_predicate<Predicate, const T&>
136 {
137 // Get current (head) node.
138 const node* current = &head_;
139 // Lock current mutex.
140 reader_lock current_lock(head_.mutex_);
141
142 // Get next node
143 while (current->next_) {
144 // If there is a next node, lock next mutex. Else, end.
145 reader_lock next_lock(current->next_->mutex_);
146 // Check the next node's value against the predicate. If predicate passes, return true.
147 if (std::invoke(std::forward<Predicate>(_predicate), *current->next_->value_)) {
148 return true;
149 }
150 // If predicate fails, advance current node and current lock.
151 current = current->next_.get();
152 current_lock = std::move(next_lock);
153 }
154
155 // Return false if none of the values passes the predicate.
156 return false;
157 }

192 /**
193 * Remove values in the list that passes the predicate.
194 * @tparam Predicate The typename of the predicate.
195 * @param _predicate Predicate to test if a value should be removed.
196 * @param _limit The maximum number of values to remove.
197 * @return The number of values removed.
198 */
199 template<class Predicate>
200 size_t remove_if(Predicate&& _predicate, size_t _limit = SIZE_MAX)
201 requires mkr::is_predicate<Predicate, const T&>
202 {
203 // Remove counter.
204 size_t num_removed = 0;
205 // Get current (head) node.
206 node* current = &head_;
207 // Lock current (head) mutex.
208 writer_lock current_lock(current->mutex_);
209
210 // Get next node
211 while (_limit && current->next_) {
212 // If there is a next node, lock next mutex. Else, end.
213 writer_lock next_lock(current->next_->mutex_);
214
215 // If predicate passes, advance next node, and discard the "old" next node.
216 if (std::invoke(std::forward<Predicate>(_predicate), *current->next_->value_)) {
217 // Point to the node we want to remove so that it does not go out of scope until we are done.
218 std::unique_ptr<node> node_to_remove = std::move(current->next_);
219 // Advance next node.
220 current->next_ = std::move(node_to_remove->next_);
221 // Unlock the mutex before node_to_remove's mutex goes out of scope and is deleted.
222 next_lock.unlock();
223 // Decrease element count.
224 --num_elements_;
225 // Increase remove counter.
226 num_removed++;
227 // Decrease limit counter.
228 --_limit;
229 continue;
230 }
231
232 // If predicate fails, advance current node & lock.
233 current = current->next_.get();
234 current_lock = std::move(next_lock);
235 }
236
237 return num_removed;
238 }
