9  /**
10  * Handle to a threadsafe exclusive read, concurrent write guarded data.
11  * @tparam L The typename of the mutex lock.
12  * @tparam V The typename of the value.
13  * @warning Do not hold on to a guard_handle for long periods.
14  * As long as the handle exist, it is LOCKING AND BLOCKING ANOTHER THREAD!
15  * DO NOT LOCK THE SAME OBJECT MULTIPLE TIMES AT THE SAME TIME IN THE SAME THREAD.
16  * Locking a mutex that the current thread already owns is UNDEFINED BEHAVIOUR.
17  * If the guard_handle exists in the same scope as the rw_guard it is from, call release() before they both
18  * go out of scope. Because there is a chance that the rw_guard gets deleted before guard_handle, and guard_handle will
19  * try to release a deleted mutex, which is UNDEFINED BEHAVIOUR.
20  *
21  * GOOD:
22  * @code
23  * void bar(rw_guard<int>* _guarded) {
24  * // The wh will go out of scope and delete BEFORE _guarded.
25  * rw_guard<int>::write_handle wh = _guarded->write_lock();
26  *
27  * // Do stuff...
28  * }
29  *
30  * void foo() {
31  * rw_guard<int> guarded;
32  * bar(&guarded);
33  * }
34  *
35  * void baz() {
36  * rw_guard<int> guarded;
37  * rw_guard<int>::write_handle wh = guarded.write_lock();
38  * // The wh releases the mutex lock before it and guarded goes out of scope.
39  * wh.release();
40  * }
41  * @endcode
42  *
43  * BAD (Like, REALLY REALLY REALLY REALLY BAD):
44  * @code
45  * void foo() {
46  * // If guarded is destroyed before wh when they go out of scope, you're FUCKED!
47  * rw_guard<int> guarded;
48  * rw_guard<int>::write_handle wh = guarded.write_lock();
49  * }
50  *
51  * void bar() {
52  * rw_guard<int> guarded;
53  * {
54  * // Locking the same mutex twice in the same thread is UNDEFINED BEHAVIOUR.
55  * rw_guard<int>::write_handle wh = guarded.write_lock();
56  * rw_guard<int>::read_handle rh = guarded.read_lock();
57  * }
58  * }
59  *
60  * class my_class
61  * {
62  * rw_guard<int>::write_handle* handle_that_never_dies;
63  *
64  * void baz()
65  * {
66  * // You are blocking every other thread out there. Prepare for deadlock.
67  * handle_that_never_dies = &guarded.write_lock();
68  * }
69  * }
70  * @endcode
71  */
