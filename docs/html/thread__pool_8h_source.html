30  /**
31  * A work stealing thread pool. Tasks can be submitted to it to be done concurrently.
32  * Once a thread is working on a task, it is not interruptable until the task is complete.
33  */
34  class thread_pool {
35  private:
36  /// The number of threads in the thread pool. The number of threads must be >= 1.
37  const size_t num_threads_;
38  /// A flag to signal the threads to start working on tasks.
39  std::atomic_int start_flag_; // TODO: Change to std::latch once Ubuntu releases G++ 10
40  // std::latch start_flag_;
41  /// A flag to signal the threads to stop after completing their current task.
42  std::atomic_bool end_flag_;
43 
44  /// A hashtable mapping the std::thread::id of a thread to the local array index.
45  threadsafe_hashtable<std::thread::id, size_t> worker_index_lookup_;
46  /// The global task queue shared by all threads. It is a FIFO queue.
47  threadsafe_queue<task> global_task_queue_;
48  /**
49  * An array of local task queues of the worker threads. It is a LIFO stack.
50  * It is best that each worker thread adds the task to it's own stack.
51  * A stack container is used instead of a queue container, because when a worker
52  * just submits a task, there is a higher chance that the memory that the task will
53  * access is still warm in the thread's cache. Therefore there is the least
54  * chance of a cache miss. That's why a stack which is LIFO is preferred.
55  */
56  std::vector<std::shared_ptr<threadsafe_stack<task>>> local_task_queues_;
57  /// An array of worker threads.
58  std::vector<std::thread> worker_threads_;
59

62  * @param _index The thread's array index.

64  */

112  /**
113  * Run a pending task in the thread pool. After submitting a task to the thread pool, make sure
114  * run pending tasks in a while loop if the thread is idle and waiting for the submitted task,
115  * especially in recursive functions. In a recursive function, it is very possible that all of the
116  * threads are waiting for a submitted task, but none of them is completing any because their current
117  * task is to wait for another task.
118  * @return Returns true if a task was run. Else, return false.
119  * Just because run_pending_task() returns false now, does not mean that another task won't
120  * submitted by another thread soon.
121  */
122  bool run_pending_task();

124  /**
125  * Submit a task to the thread pool.
126  * @tparam Callable The typename of the function or callable object.
127  * @tparam Args The typename of the function arguments.
128  * @param _func The function or callable object.
129  * @param _args The function arguments.
130  * @return A std::future which will contain the result of the function.
131  */
132  template<typename Callable, typename... Args>
133  std::future<std::invoke_result_t<Callable, Args...>> submit(Callable&& _func, Args&& ... _args)
134  {
135  typedef std::invoke_result_t<Callable, Args...> result_t;
136 
137  // C++ 20 perfect capture. [https://stackoverflow.com/questions/47496358/c-lambdas-how-to-capture-variadic-parameter-pack-from-the-upper-scope]
138  std::packaged_task<result_t(void)> p_task{
139  [func = std::forward<Callable>(_func),
140  ...args = std::forward<Args>(_args)]() {
141  return std::invoke(func, args...);
142  }};
143  std::future<result_t> result = p_task.get_future();
144 
145  // Get the worker index of this thread.
146  // If the task was submitted from a non worker thread, it will not have a worker index.
147  // In that case, add the task to the global queue.
148  std::shared_ptr<size_t> worker_index = worker_index_lookup_.at(std::this_thread::get_id());
149  if (worker_index) {
150  local_task_queues_[*worker_index]->push(task{std::move(p_task)});
151  }
152  else {
153  global_task_queue_.push(task{std::move(p_task)});
154  }
155 
156  return result;
157  }
158  };
159 }
