14  /**
15  * Threadsafe exclusive read, concurrent write guard.
16  *
17  * Addtional Requirements:
18  * - rw_guard must be able to support type T where T is a non-copyable or non-movable type.
19  * - rw_guard does not have to support a non-copyable AND non-movable type.
20  *
21  * @tparam T The typename of the contained value.
22  */
23  template<typename T>
24  class rw_guard {
25  private:
26  typedef std::shared_timed_mutex mutex_type;
27  typedef std::unique_lock<mutex_type> writer_lock;
28  typedef std::shared_lock<mutex_type> reader_lock;
29 
30  mutable mutex_type value_mutex_;
31  std::unique_ptr<T> value_;
32 
33  public:
34  typedef guard_handle<writer_lock, T> write_handle;
35  typedef guard_handle<reader_lock, const T> read_handle;
36 
37  /**
38  * Constructs the rw_guard.
39  * @tparam Args Typename of the parameters of T's constructor.
40  * @param _args Parameters of T's constructor.
41  */
42  template<typename... Args>
43  explicit rw_guard(Args&& ... _args)
44  : value_(std::make_unique<T>(std::forward<Args>(_args)...)) { }
45 
46  /**
47  * Destructs the rw_guard.
48  */
49  ~rw_guard() { }

56  /**
57  * Lock the data for writing.
58  * @return A write_handle for writing to the value.
59  * @warning Avoid obtaining multiple locks at the same time, or holding on to locks for long periods of time.
60  * Locks are BLOCKING and may stall other threads until the lock is released.
61  * It is also dangerous to obtain multiple locks on the same thread. If the same thread was to lock a mutex
62  * it already owned again, it causes UNDEFINED BEHAVIOUR.
63  */
64  [[nodiscard]]
65  write_handle write_lock()
66  {
67  return write_handle(writer_lock(value_mutex_), value_.get());
68  }
69 
70  /**
71  * Try to lock the data for writing.
72  * @return A write_handle for writing to the value. If another thread already locked the data, an empty handle is returned.
73  * @warning Avoid obtaining multiple locks at the same time, or holding on to locks for long periods of time.
74  * Locks are BLOCKING and may stall other threads until the lock is released.
75  * It is also dangerous to obtain multiple locks on the same thread. If the same thread was to lock a mutex
76  * it already owned again, it causes UNDEFINED BEHAVIOUR.
77  */
78  [[nodiscard]]
79  write_handle try_write_lock()
80  {
81  writer_lock lock(value_mutex_, std::try_to_lock);
82  if (lock.owns_lock()) {
83  return write_handle(std::move(lock), value_.get());
84  }
85  return write_handle(std::move(lock), nullptr);
86  }

89  * Lock the data for reading.
90  * @return A read_handle for reading from the value.
91  * @warning Avoid obtaining multiple locks at the same time, or holding on to locks for long periods of time.
92  * Locks are BLOCKING and may stall other threads until the lock is released.
93  * It is also dangerous to obtain multiple locks on the same thread. If the same thread was to lock a mutex
94  * it already owned again, it causes UNDEFINED BEHAVIOUR.
95  */
96  [[nodiscard]]
97  read_handle read_lock() const
98  {
99  return read_handle(reader_lock(value_mutex_), value_.get());
100  }

102  /**
103  * Try to lock the data for reading.
104  * @return A read_handle for reading from the value. If another thread already locked the data for writing, an empty handle is returned.
105  * @warning Avoid obtaining multiple locks at the same time, or holding on to locks for long periods of time.
106  * Locks are BLOCKING and may stall other threads until the lock is released.
107  * It is also dangerous to obtain multiple locks on the same thread. If the same thread was to lock a mutex
108  * it already owned again, it causes UNDEFINED BEHAVIOUR.
109  */
110  [[nodiscard]]
111  read_handle try_read_lock() const
112  {
113  reader_lock lock(value_mutex_, std::try_to_lock);
114  if (lock.owns_lock()) {
115  return read_handle(std::move(lock), value_.get());
116  }
117  return read_handle(std::move(lock), nullptr);
118  }
