11  /**
12  * Threadsafe hashtable.
13  *
14  * Invariants:
15  * - The number of buckets do not change.
16  * - Each key can only appear once in the hashtable at any given time.
17  * - The same key will always be mapped to the same bucket.
18  * - Each key can only be mapped to a single value.
19  * - A key must have a value.
20  * - When iterating from the head_ of a bucket, it will eventually lead to the tail.
21  * - If the corresponding bucket does not contain the key, it does not exist.
22  * - The same key must always return the same hash.
23  *
24  * Addtional Requirements:
25  * - threadsafe_hashtable must be able to support type V where V is a non-copyable or non-movable type.
26  * - threadsafe_hashtable does not have to support a non-copyable AND non-movable type.
27  *
28  * @tparam K The typename of the key.
29  * @tparam V The typename of the value.
30  * @tparam N The number of buckets in the hashtable. Prime numbers are highly recommended.
31  */
32  template<typename K, typename V, std::size_t N = 47>

98  /**
99  * Internal function to add a key-value pair into the hashtable.
100  * @param _key The key to add.
101  * @param _value The value to add.
102  * @return Returns true if key-value pair was added successfully. Returns false if key already exists in the hashtable.
103  */
104  bool do_insert(const K& _key, std::shared_ptr<V> _value)
105  {
106  std::function<bool(const pair&)> match_key = [&_key](const pair& _pair) { return _pair.match_key(_key); };
107 
108  // Lock bucket so that no other threads can add to the bucket at the same time.
109  bucket& b = get_bucket(_key);
110  writer_lock b_lock(b.mutex_);
111 
112  // If the bucket does not contain they key, add the new pair to the bucket and return true.
113  if (b.list_.match_none(match_key)) {
114  b.list_.push_front(pair(_key, _value));
115  ++num_elements_;
116  return true;
117  }
118 
119  // Otherwise, if the bucket already contains the key, return false.
120  return false;
121  }

123  /**
124  * Internal function to replace an existing key-value pair in the hashtable.
125  * @param _key The key to replace.
126  * @param _value The new value to replace the old value.
127  * @return Returns true if key-value pair was replaced successfully. Returns false if key does not already exists in the hashtable.
128  */
129  bool do_replace(const K& _key, std::shared_ptr<V> _value)
130  {
131  std::function<bool(const pair&)> match_key = [&_key](const pair& _pair) { return _pair.match_key(_key); };
132  std::function<pair(void)> supplier = [_key, _value](void) { return pair(_key, _value); };
133 
134  // Lock bucket so that no other threads can add to the bucket at the same time.
135  bucket& b = get_bucket(_key);
136  writer_lock b_lock(b.mutex_);
137 
138  // Return true if the pair was successfully replace, otherwise, return false.
139  return b.list_.replace_if(match_key, supplier);
140  }
