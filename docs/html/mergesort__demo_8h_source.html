18  template<typename T>
19  static void do_sort(T* _array, T* _temp_buffer, int _start, int _mid, int _end)
20  {
21  int left_index = _start;
22  int right_index = _mid;
23 
24  for (int i = _start; i<_end; i++) {
25  if (left_index==_mid) {
26  _temp_buffer[i] = _array[right_index++];
27  }
28  else if (right_index==_end) {
29  _temp_buffer[i] = _array[left_index++];
30  }
31  else if (_array[left_index]<_array[right_index]) {
32  _temp_buffer[i] = _array[left_index++];
33  }
34  else {
35  _temp_buffer[i] = _array[right_index++];
36  }
37  }
38 
39  for (int i = _start; i<_end; i++) {
40  _array[i] = _temp_buffer[i];
41  }
42  }

44  template<typename T>
45  static void single_thread_mergesort(T* _array, T* _temp_buffer, int _start, int _end)
46  {
47  int num_elements = _end-_start;
48  if (num_elements<=1) { return; }
49 
50  int mid = _start+((num_elements) >> 1);
51  single_thread_mergesort(_array, _temp_buffer, _start, mid);
52  single_thread_mergesort(_array, _temp_buffer, mid, _end);
53 
54  do_sort(_array, _temp_buffer, _start, mid, _end);
55  }

57  template<typename T>
58  static void
59  thread_pool_mergesort(T* _array, T* _temp_buffer, int _start, int _end, thread_pool* _thread_pool,
60  int _granularity)
61  {
62  int num_elements = _end-_start;
63  if (num_elements<=1) { return; }
64 
65  int mid = _start+(num_elements >> 1);
66 
67  // If num_elements < _granularity, it is not worth the overhead to spawn a new task.
68  std::future<void> fork;
69  if (num_elements>=_granularity) {
70  fork = _thread_pool->submit([_array, _temp_buffer, _start, mid, _thread_pool, _granularity]() {
71  thread_pool_mergesort(_array, _temp_buffer, _start, mid, _thread_pool, _granularity);
72  });
73  }
74  else {
75  thread_pool_mergesort(_array, _temp_buffer, _start, mid, _thread_pool, _granularity);
76  }
77 
78  thread_pool_mergesort(_array, _temp_buffer, mid, _end, _thread_pool, _granularity);
79 
80  // If num_elements < _granularity, it is not worth the overhead to spawn a new task.
81  if (num_elements>=_granularity) {
82  /* If fork isn't ready, we run pending tasks on the thread pool.
83  * If we do not run any pending task, since this function is recursive, there is a high
84  * chance that all the threads may end up simply waiting for fork.get() and no threads
85  * are actually doing any work, resulting in a deadlock.
86  */
87  while (!is_future_ready(fork)) {
88  _thread_pool->run_pending_task();
89  }
90  fork.get();
91  }
92 
93  do_sort(_array, _temp_buffer, _start, mid, _end);
94  }

96  template<typename T>
97  static void async_mergesort(T* _array, T* _temp_buffer, int _start, int _end, int _granularity)
98  {
99  int num_elements = _end-_start;
100  if (num_elements<=1) { return; }
101 
102  int mid = _start+(num_elements >> 1);
103 
104  // If num_elements < _granularity, it is not worth the overhead to spawn a new thread.
105  std::future<void> fork;
106  if (num_elements>=_granularity) {
107  fork = std::async([_array, _temp_buffer, _start, mid, _granularity]() -> void {
108  async_mergesort(_array, _temp_buffer, _start, mid, _granularity);
109  });
110  }
111  else {
112  async_mergesort(_array, _temp_buffer, _start, mid, _granularity);
113  }
114  async_mergesort(_array, _temp_buffer, mid, _end, _granularity);
115 
116  // If num_elements < _granularity, it is not worth the overhead to spawn a new thread.
117  if (num_elements>=_granularity) { fork.get(); }
118 
119  do_sort(_array, _temp_buffer, _start, mid, _end);
120  }

131  /**
132  * Call this function to run the merge sort demo.
133  * @param _num_loops The number of times each test (Single Thread, mkr::thread_pool, std::async) runs. The more times a test run, the more accurate the average time.
134  * @param _array_size The number of elements to sort.
135  * @param _granularity For mkr::thread_pool and std::async, a new task will be submitted to the thread pool (mkr::thread_pool), or a new thread will only be created (std::async),
136  * when the number of elements to sort is greater than this number. There is an inherent cost to creating a new thread or submitting a task.
137  * Therefore, if there is only a small number of elements to sort, it may be cheaper to just run it all in the current thread than to split half of it to another thread.
138  * @param _display_unsorted Display the unsorted array.
139  * @param _display_sorted Display the sorted array.
