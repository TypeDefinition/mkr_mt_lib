14  /**
15  * Threadsafe stack.
16  *
17  * Invariants:
18  * - top_ == nullptr means that the stack is empty.
19  * - Traversing top_->next_ will eventually lead to the bottom node.
20  *
21  * Addtional Requirements:
22  * - threadsafe_stack must be able to support type T where T is a non-copyable or non-movable type.
23  * - threadsafe_stack does not have to support a non-copyable AND non-movable type.
24  *
25  * Additional Notes:
26  * - threadsafe_stack is non-copyable AND non-movable. (As it has non-copyable and non-movable members.)
27  *
28  * @tparam T The typename of the contained values.
29  */
30  template<typename T>
31  class threadsafe_stack {
32  private:
33  typedef std::timed_mutex mutex_type;
34 
35  /**
36  * A node containing a value, and the next node in the stack.
37  */
38  struct node {
39  /// Value.
40  std::shared_ptr<T> value_;
41  /// Next node.
42  std::unique_ptr<node> next_;
43  };
44 
45  /// Mutex of the top of the stack.
46  mutable mutex_type top_mutex_;
47  /// Condition variable to notify waiting threads of a push.
48  std::condition_variable_any cond_;
49  /// Top node of the stack.
50  std::unique_ptr<node> top_;
51  /// Number of elements in the stack.
52  std::atomic_size_t num_elements_;
53 
54  /**
55  * Internal function to push a new value onto the top of the stack.
56  * @param _value The value to push onto the top of the stack.
57  * @warning If multiple threads pushes at the same time, the value will be inserted but may no longer be at the top of the stack.
58  */
59  void do_push(std::shared_ptr<T> _value)
60  {
61  // Construct a new node.
62  std::unique_ptr<node> new_node = std::make_unique<node>();
63  // Set the new node's value to the new value.
64  new_node->value_ = _value;
65  // Lock the top mutex.
66  std::lock_guard<mutex_type> lock(top_mutex_);
67  // Set the new node's next node to the old top node.
68  new_node->next_ = std::move(top_);
69  // Set the new node as the top node.
70  top_ = std::move(new_node);
71  // Increase the element counter.
72  ++num_elements_;
73  }
74 
75  /**
76  * Internal function to pop a value from the top of the stack.
77  * @return Pops and return the value from the top of the stack.
78  */
79  std::shared_ptr<T> do_pop()
80  {
81  // Extract the value of the top node.
82  std::shared_ptr<T> head_data = top_->value_;
83  // Point the top node to it's next node.
84  top_ = std::move(top_->next_);
85  // Decrease the element counter.
86  --num_elements_;
87  // Return the extracted value.
88  return head_data;
89  }
90 
91  public:
92  /**
93  * Constructs the stack.
94  */
95  threadsafe_stack()
96  :num_elements_(0) { }
97 
98  /**
99  * Destructs the stack.
100  */
101  ~threadsafe_stack() = default;

108  /**
109  * Push a new value to the top of the stack.
110  * @param _value The value to push to the top of the stack.
111  * @warning If multiple threads pushes at the same time, the value will be inserted but may no longer be at the top of the stack.
112  */
113  void push(const T& _value)
114  {
115  // When possible, expensive operations like constructing an object should be done before acquiring the mutex.
116  do_push(std::make_shared<T>(_value));
117  // Notify any threads waiting for value_. This is done AFTER unlocking the mutex so that any waiting threads can operate immediately.
118  cond_.notify_one();
119  }
120 
121  /**
122  * Push a new value to the top of the stack.
123  * @param _value The value to push to the top of the stack.
124  * @warning If multiple threads pushes at the same time, the value will be inserted but may no longer be at the top of the stack.
125  */
126  void push(T&& _value)
127  {
128  // When possible, expensive operations like constructing an object should be done before acquiring the mutex.
129  do_push(std::make_shared<T>(std::forward<T>(_value)));
130  // Notify any threads waiting for value_. This is done AFTER unlocking the mutex so that any waiting threads can operate immediately.
131  cond_.notify_one();
132  }

134  /**
135  * Wait and remove the top value from the stack.
136  * @return Pops and return the value from the top of the stack. If the stack is empty, wait until another thread pushes a value.
137  */
138  std::shared_ptr<T> wait_and_pop()
139  {
140  // Lock the top mutex.
141  std::unique_lock<mutex_type> lock(top_mutex_);
142  cond_.wait(lock, [this]() { return top_!=nullptr; });
143  return do_pop();
144  }
145 
146  /**
147  * Try to remove the top value from the stack.
148  * @return Pop and return the value from the top of the stack. If the stack is empty, return a nullptr.
149  */
150  std::shared_ptr<T> try_pop()
151  {
152  // Lock the top mutex.
153  std::lock_guard<mutex_type> lock(top_mutex_);
154  if (top_==nullptr) { return nullptr; }
155  return do_pop();
156  }
157 
158  /**
159  * Clears the stack.
160  */
161  void clear()
162  {
163  // Lock the mutex and prevent anyone from pushing.
164  std::lock_guard<mutex_type> lock(top_mutex_);
165  // Pop all nodes.
166  while (top_!=nullptr) { do_pop(); }
167  }

170  * Checks if the container is empty.
171  * @return Returns true if the container is empty, false otherwise.
172  */
173  bool empty() const { return num_elements_.load()==0; }
174 
175  /**
176  * Returns the number of elements in the container.
177  * @return Returns the number of elements in the container.
178  */
179  size_t size() const { return num_elements_.load(); }
180  };
