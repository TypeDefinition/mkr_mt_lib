11  /**
12  * A task is used when there is a need to point to callable objects of different types. It is able to store the callable objects using type erasure.
13  * One such usage is in mkr::thread_pool, where a queue of task is used to point to the various functors submitted to the thread pool.
14  */
15  class task {
16  private:
17  class wrapper_interface {
18  public:
19  wrapper_interface() = default;
20  virtual ~wrapper_interface() = default;
21  virtual void operator()() = 0;
22  };
23 
24  template<typename Callable>
25  class templated_wrapper : public wrapper_interface {
26  private:
27  Callable func_;
28  public:
29  explicit templated_wrapper(Callable&& _func)
30  :func_(std::forward<Callable>(_func)) { }
31  virtual ~templated_wrapper() = default;
32  virtual void operator()() { func_(); }
33 
34  // Callable might either be non-movable or non-copyable. So template_wrapper supports neither.
35  // Once created, a templated_wrapper should not be moved or copied.
36  templated_wrapper(templated_wrapper&&) = delete;
37  templated_wrapper& operator=(templated_wrapper&&) = delete;
38  templated_wrapper(const templated_wrapper&) = delete;
39  templated_wrapper& operator=(const templated_wrapper&) = delete;
40  };
41 
42  /// The wrapped callable. We cannot move or copy wrapper_interface, but we can move the pointer to wrapper interface.
43  std::unique_ptr<wrapper_interface> function_ptr_;
44 
45  public:
46  /**
47  * Constructs the task.
48  * @tparam Callable The callable to invoke.
49  * @param _func The callable object.
50  */
51  template<typename Callable>
52  explicit task(Callable&& _func)
53  {
54  // By using std::forward, the callable will be constructed via move constructor or copy constructor accordingly.
55  function_ptr_ = std::make_unique<templated_wrapper<Callable>>(std::forward<Callable>(_func));
56  }
57 
58  /**
59  * Move constructor.
60  * @param _other The other task to move from.
61  * @attention This constructor must be specified, otherwise calling the move constructor will result in the templated constructor being constructed.
62  * That will result in a task that creates and calls a task, which creates and calls a task, which creates and calls a task, which creates and calls...
63  */
64  task(task&& _other) noexcept
65  {
66  function_ptr_ = std::move(_other.function_ptr_);
67  }
68 
69  /**
70  * Destructs the task.
71  */
72  ~task() = default;
73 
74  /**
75  * Move assignment operator.
76  * @param _other The other task to move from.
77  * @return This task after the move.
78  */
79  task& operator=(task&& _other) noexcept
80  {
81  function_ptr_ = std::move(_other.function_ptr_);
82  return *this;
83  }
84 
85  /**
86  * Invoke the wrapped callable object.
87  */
88  void operator()() { function_ptr_->operator()(); }
89 
90  task(const task&) = delete;
91  task& operator=(const task&) = delete;
92  };
