8  {

12  local_task_queues_.push_back(std::make_shared<threadsafe_stack<task>>());
13  std::thread worker_thread{&thread_pool::worker_thread_func, this};
14  worker_index_lookup_.insert(worker_thread.get_id(), i);
15  worker_threads_.emplace_back(std::move(worker_thread));
