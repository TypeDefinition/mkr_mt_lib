140  /**
141  * Submit a task to the thread pool.
142  * @tparam Callable The typename of the function or callable object.
143  * @tparam Args The typename of the function arguments.
144  * @param _func The function or callable object.
145  * @param _args The function arguments.
146  * @return A std::future which will contain the result of the function.
147  */
148  template<typename Callable, typename... Args>
149  std::future<std::invoke_result_t<Callable, Args...>> submit(Callable&& _func, Args&& ... _args)
150  {
151  typedef std::invoke_result_t<Callable, Args...> result_t;
152 
153  // C++ 20 perfect capture. [https://stackoverflow.com/questions/47496358/c-lambdas-how-to-capture-variadic-parameter-pack-from-the-upper-scope]
154  std::packaged_task<result_t(void)> p_task{
155  [func = std::forward<Callable>(_func),
156  ...args = std::forward<Args>(_args)]() {
157  return std::invoke(func, args...);
158  }};
159  std::future<result_t> result = p_task.get_future();
160 
161  // Get the worker index of this thread.
162  // If the task was submitted from a non worker thread, it will not have a worker index.
163  // In that case, add the task to the global queue.
164  std::shared_ptr<size_t> worker_index = worker_index_lookup_.get(std::this_thread::get_id());
165  if (worker_index) {
166  local_task_queues_[*worker_index]->push(task{std::move(p_task)});
167  }
168  else {
169  global_task_queue_.push(task{std::move(p_task)});
170  }
171 
172  return result;
173  }
